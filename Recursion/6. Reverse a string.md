## Return the original string
```cpp
#include <bits/stdc++.h>
using namespace std;

void reverse(string& s, int i, int j)
{
    if(i > j)
        return ;

    swap(s[i], s[j]);
    i++;
    j--;
    reverse(s, i, j);
}

int main() {
    string s = "abcde";
    int length = s.length();
    reverse(s, 0, length - 1);
    cout << s;
}
```

## Return a new string
```cpp
#include <bits/stdc++.h>
using namespace std;

string reverse(string str, int len, int index = 0) {
    if(index > len - 1)
        return "" + str[index];

    return reverse(str, len, index + 1) + str[index];
}


int main() {
    string str = "Hello World";
    
    cout << reverse(str, str.size());
    return 0;
}
```
